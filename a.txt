## `permissions/repository.py` - DB 접근 계층

```python
from sqlalchemy.orm import Session
from sqlalchemy import exists
from models import UserPermission, Permission


class PermissionRepository:
    def __init__(self, db: Session):
        self.db = db

    def has_any_permission(self, user_id: int) -> bool:
        """유저에게 매핑된 권한이 하나라도 있는지 확인"""
        return self.db.query(
            exists().where(UserPermission.user_id == user_id)
        ).scalar()

    def has_specific_permission(self, user_id: int, permission_code: str) -> bool:
        """유저가 특정 권한을 보유하는지 확인"""
        return self.db.query(
            exists()
            .where(UserPermission.user_id == user_id)
            .where(UserPermission.permission_id == Permission.id)
            .where(Permission.code == permission_code)
        ).scalar()

    def get_user_permissions(self, user_id: int) -> list[Permission]:
        """유저의 전체 권한 목록 조회"""
        mappings = (
            self.db.query(UserPermission)
            .filter(UserPermission.user_id == user_id)
            .all()
        )
        return [m.permission for m in mappings]

    def grant_permission(self, user_id: int, permission_id: int) -> UserPermission:
        """권한 부여 (중복 방지)"""
        existing = (
            self.db.query(UserPermission)
            .filter_by(user_id=user_id, permission_id=permission_id)
            .first()
        )
        if existing:
            return existing

        mapping = UserPermission(user_id=user_id, permission_id=permission_id)
        self.db.add(mapping)
        self.db.commit()
        self.db.refresh(mapping)
        return mapping

    def revoke_permission(self, user_id: int, permission_id: int) -> bool:
        """권한 회수"""
        mapping = (
            self.db.query(UserPermission)
            .filter_by(user_id=user_id, permission_id=permission_id)
            .first()
        )
        if not mapping:
            return False
        self.db.delete(mapping)
        self.db.commit()
        return True
```

## `permissions/service.py` - 비즈니스 로직 계층

```python
from sqlalchemy.orm import Session
from fastapi import HTTPException, status
from .repository import PermissionRepository


class PermissionService:
    def __init__(self, db: Session):
        self.repo = PermissionRepository(db)

    def check_has_any_permission(self, user_id: int) -> dict:
        """
        유저에게 매핑된 권한이 하나라도 있는지 체크
        - 없으면 403 반환
        """
        has_permission = self.repo.has_any_permission(user_id)

        if not has_permission:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail=f"유저(id={user_id})에게 매핑된 권한이 없습니다.",
            )

        return {
            "user_id": user_id,
            "has_any_permission": True,
        }

    def check_specific_permission(self, user_id: int, permission_code: str) -> dict:
        """특정 권한 코드 보유 여부 체크"""
        # 먼저 권한 자체가 하나라도 있는지 확인
        if not self.repo.has_any_permission(user_id):
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail=f"유저(id={user_id})에게 매핑된 권한이 없습니다.",
            )

        has_permission = self.repo.has_specific_permission(user_id, permission_code)
        return {
            "user_id": user_id,
            "permission_code": permission_code,
            "has_permission": has_permission,
        }

    def get_user_permissions(self, user_id: int) -> dict:
        """유저의 전체 권한 목록 반환"""
        permissions = self.repo.get_user_permissions(user_id)
        return {
            "user_id": user_id,
            "permissions": permissions,
            "total": len(permissions),
        }

    def grant_permission(self, user_id: int, permission_id: int) -> dict:
        """권한 부여"""
        mapping = self.repo.grant_permission(user_id, permission_id)
        return {
            "message": "권한이 부여되었습니다.",
            "mapping_id": mapping.id,
        }

    def revoke_permission(self, user_id: int, permission_id: int) -> dict:
        """권한 회수"""
        success = self.repo.revoke_permission(user_id, permission_id)
        if not success:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail="해당 권한 매핑을 찾을 수 없습니다.",
            )
        return {"message": "권한이 회수되었습니다."}
```

## `permissions/router.py` - 라우터 계층

```python
from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from database import get_db
from schemas import GrantPermissionRequest, CheckPermissionRequest
from .service import PermissionService

router = APIRouter(prefix="/permissions", tags=["permissions"])


def get_permission_service(db: Session = Depends(get_db)) -> PermissionService:
    return PermissionService(db)


@router.get("/user/{user_id}/has-any")
def check_has_any_permission(
    user_id: int,
    service: PermissionService = Depends(get_permission_service),
):
    """유저에게 매핑된 권한이 하나라도 있는지 체크"""
    return service.check_has_any_permission(user_id)


@router.post("/check")
def check_specific_permission(
    req: CheckPermissionRequest,
    service: PermissionService = Depends(get_permission_service),
):
    """특정 권한 코드 보유 여부 확인"""
    return service.check_specific_permission(req.user_id, req.permission_code)


@router.get("/user/{user_id}")
def get_user_permissions(
    user_id: int,
    service: PermissionService = Depends(get_permission_service),
):
    """유저의 전체 권한 목록 조회"""
    return service.get_user_permissions(user_id)


@router.post("/grant")
def grant_permission(
    req: GrantPermissionRequest,
    service: PermissionService = Depends(get_permission_service),
):
    """권한 부여"""
    return service.grant_permission(req.user_id, req.permission_id)


@router.delete("/revoke")
def revoke_permission(
    req: GrantPermissionRequest,
    service: PermissionService = Depends(get_permission_service),
):
    """권한 회수"""
    return service.revoke_permission(req.user_id, req.permission_id)
```

## 계층별 역할 요약

```
Router  → HTTP 요청/응답만 담당, Service 호출
Service → 비즈니스 로직 (권한 없으면 403 등 예외 처리)
Repo    → DB 쿼리만 담당, exists()로 최적화된 존재 여부 확인
```

**핵심 포인트** 두 가지:

1. **`has_any_permission`** — `SELECT EXISTS(...)` 쿼리를 사용해 전체 조회 없이 존재 여부만 효율적으로 확인
2. **의존성 주입** — `Depends(get_permission_service)`로 Router → Service → Repository 체인을 FastAPI DI로 깔끔하게 연결
