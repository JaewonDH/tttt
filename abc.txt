## `permissions/repository.py` - 유저 정보 조회 추가

```python
from sqlalchemy.orm import Session, joinedload
from sqlalchemy import exists
from models import UserPermission, Permission, User


class PermissionRepository:
    def __init__(self, db: Session):
        self.db = db

    def has_any_permission(self, user_id: int) -> bool:
        return self.db.query(
            exists().where(UserPermission.user_id == user_id)
        ).scalar()

    def get_user_with_permissions(self, user_id: int) -> User | None:
        """유저 정보 + 매핑된 권한 목록을 한 번의 쿼리로 조회"""
        return (
            self.db.query(User)
            .options(
                joinedload(User.permission_mappings)   # UserPermission 조인
                .joinedload(UserPermission.permission) # Permission 조인
            )
            .filter(User.id == user_id)
            .first()
        )
```

## `schemas.py` - 응답 스키마 추가

```python
from pydantic import BaseModel
from datetime import datetime
from typing import List


class PermissionResponse(BaseModel):
    id: int
    code: str
    description: str | None = None

    class Config:
        from_attributes = True


class UserPermissionCheckResponse(BaseModel):
    """has_any_permission 체크 성공 시 반환 스키마"""
    user_id: int
    username: str
    email: str
    is_active: bool
    created_at: datetime
    has_any_permission: bool
    permissions: List[PermissionResponse]
    permission_count: int

    class Config:
        from_attributes = True
```

## `permissions/service.py` - 유저 정보 포함해서 반환

```python
from sqlalchemy.orm import Session
from fastapi import HTTPException, status
from schemas import UserPermissionCheckResponse, PermissionResponse
from .repository import PermissionRepository


class PermissionService:
    def __init__(self, db: Session):
        self.repo = PermissionRepository(db)

    def check_has_any_permission(self, user_id: int) -> UserPermissionCheckResponse:
        """
        권한이 하나라도 있는지 확인 후
        유저 정보 + 전체 권한 목록 반환
        """
        # 1. 유저 정보 + 권한 한 번에 조회
        user = self.repo.get_user_with_permissions(user_id)

        if not user:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail=f"유저(id={user_id})를 찾을 수 없습니다.",
            )

        # 2. 매핑된 권한 추출
        permissions = [m.permission for m in user.permission_mappings]

        # 3. 권한이 하나도 없으면 403
        if not permissions:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail=f"유저(id={user_id})에게 매핑된 권한이 없습니다.",
            )

        # 4. 유저 정보 + 권한 목록 반환
        return UserPermissionCheckResponse(
            user_id=user.id,
            username=user.username,
            email=user.email,
            is_active=user.is_active,
            created_at=user.created_at,
            has_any_permission=True,
            permissions=[
                PermissionResponse(
                    id=p.id,
                    code=p.code,
                    description=p.description,
                )
                for p in permissions
            ],
            permission_count=len(permissions),
        )
```

## `permissions/router.py` - 응답 타입 명시

```python
from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from database import get_db
from schemas import UserPermissionCheckResponse
from .service import PermissionService

router = APIRouter(prefix="/permissions", tags=["permissions"])


def get_permission_service(db: Session = Depends(get_db)) -> PermissionService:
    return PermissionService(db)


@router.get(
    "/user/{user_id}/has-any",
    response_model=UserPermissionCheckResponse,
)
def check_has_any_permission(
    user_id: int,
    service: PermissionService = Depends(get_permission_service),
):
    """유저 권한 존재 여부 확인 + 유저 전체 정보 반환"""
    return service.check_has_any_permission(user_id)
```

## 핵심 변경 포인트

| 항목 | 내용 |
|------|------|
| **쿼리 최적화** | `joinedload` 로 User + UserPermission + Permission을 **쿼리 1번**에 처리 (N+1 방지) |
| **예외 분리** | 유저 없음 `404` / 권한 없음 `403` 케이스를 명확히 구분 |
| **응답 스키마** | `UserPermissionCheckResponse`로 반환 타입을 Pydantic으로 고정 |
