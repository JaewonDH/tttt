Here's a production-style refactor:

```python
from sqlalchemy import select, and_, exists
from sqlalchemy.orm import aliased
from sqlalchemy.ext.asyncio import AsyncSession

from app.exceptions import ForbiddenException  # 커스텀 예외
from app.models import Agent, CommonCode, AgentMember
from app.schemas.agent import AgentDetailResponse, MemberResponse


class AgentRepository:
    def __init__(self, session: AsyncSession):
        self.session = session

    async def _check_membership(self, agent_id: str, user_id: str) -> None:
        stmt = select(
            exists().where(
                and_(
                    AgentMember.agent_id == agent_id,
                    AgentMember.id == user_id,
                )
            )
        )
        if not await self.session.scalar(stmt):
            raise ForbiddenException("해당 Agent에 대한 접근 권한이 없습니다.")

    async def _fetch_agent(self, agent_id: str) -> dict | None:
        stat = aliased(CommonCode, name="stat")
        dept = aliased(CommonCode, name="dept")

        stmt = (
            select(
                Agent.id.label("agent_id"),
                Agent.name.label("agent_name"),
                stat.code_name.label("status_name"),
                dept.code_name.label("dept_name"),
            )
            .outerjoin(stat, and_(Agent.status_code == stat.code_value, stat.code_group == "STAT"))
            .outerjoin(dept, and_(Agent.dept_code == dept.code_value, dept.code_group == "DEPT"))
            .where(Agent.id == agent_id)
        )
        row = (await self.session.execute(stmt)).mappings().first()
        return dict(row) if row else None

    async def _fetch_members(self, agent_id: str) -> list[dict]:
        stmt = (
            select(
                AgentMember.id.label("member_id"),
                AgentMember.name.label("member_name"),
            )
            .where(AgentMember.agent_id == agent_id)
        )
        rows = (await self.session.execute(stmt)).mappings().all()
        return [dict(r) for r in rows]

    async def get_agent_detail(self, agent_id: str, user_id: str) -> AgentDetailResponse:
        await self._check_membership(agent_id, user_id)

        agent = await self._fetch_agent(agent_id)
        if agent is None:
            raise NotFoundException("Agent를 찾을 수 없습니다.")

        agent["members"] = await self._fetch_members(agent_id)
        return AgentDetailResponse(**agent)
```

주요 변경점:

**메서드 분리** — 권한 검증, Agent 조회, 멤버 조회를 private 메서드로 추출했습니다. 각각 단독 테스트가 가능하고, 다른 public 메서드에서 재사용할 수 있습니다.

**커스텀 예외** — `PermissionError` 같은 빌트인 대신 `ForbiddenException`, `NotFoundException` 같은 도메인 예외를 사용합니다. FastAPI의 exception handler에서 적절한 HTTP 상태코드(403, 404)로 매핑하기 쉽습니다.

**None 반환 제거** — `None`을 반환하면 호출하는 Service/Router 쪽에서 매번 null 체크를 해야 합니다. Repository에서 바로 예외를 던지는 게 실무에서 더 일반적입니다.

**alias에 name 부여** — 디버깅 시 SQL 로그에서 `stat_1`, `dept_1` 대신 의미 있는 이름이 보입니다.
