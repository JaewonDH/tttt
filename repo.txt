from sqlalchemy import select, and_, exists
from sqlalchemy.orm import aliased
from sqlalchemy.ext.asyncio import AsyncSession
# from models import Agent, CommonCode, AgentMember

class AgentRepository:
    def __init__(self, session: AsyncSession):
        self.session = session

    async def check_is_member(self, agent_id: str, user_id: str) -> bool:
        """해당 유저가 특정 Agent의 멤버인지 확인합니다."""
        stmt = select(
            exists().where(
                and_(
                    AgentMember.agent_id == agent_id,
                    AgentMember.id == user_id
                )
            )
        )
        return await self.session.scalar(stmt)

    async def get_agent_basic_info(self, agent_id: str) -> dict | None:
        """Agent의 기본 정보와 공통코드 명칭을 조회합니다."""
        StatCode = aliased(CommonCode)
        DeptCode = aliased(CommonCode)

        stmt = (
            select(
                Agent.id.label("agent_id"),
                Agent.name.label("agent_name"),
                StatCode.code_name.label("status_name"),
                DeptCode.code_name.label("dept_name")
            )
            .outerjoin(StatCode, and_(Agent.status_code == StatCode.code_value, StatCode.code_group == 'STAT'))
            .outerjoin(DeptCode, and_(Agent.dept_code == DeptCode.code_value, DeptCode.code_group == 'DEPT'))
            .where(Agent.id == agent_id)
        )
        
        result = await self.session.execute(stmt)
        row = result.mappings().first()
        return dict(row) if row else None

    async def get_agent_members(self, agent_id: str) -> list[dict]:
        """Agent에 소속된 멤버 목록을 조회합니다."""
        stmt = (
            select(
                AgentMember.id.label("member_id"),
                AgentMember.name.label("member_name")
            )
            .where(AgentMember.agent_id == agent_id)
        )
        
        result = await self.session.execute(stmt)
        return [dict(row) for row in result.mappings().all()]

class AgentService:
    def __init__(self, repo: AgentRepository):
        self.repo = repo

    async def get_agent_detail(self, agent_id: str, user_id: str) -> AgentDetailResponse | None:
        """권한 검증 후 Agent 상세 정보를 Pydantic 모델로 조립하여 반환합니다."""
        
        # 1. 권한 검증 (Fail-Fast)
        is_member = await self.repo.check_is_member(agent_id, user_id)
        if not is_member:
            raise PermissionError("해당 Agent에 대한 접근 권한이 없습니다.")

        # 2. Agent 기본 정보 조회
        agent_info = await self.repo.get_agent_basic_info(agent_id)
        if not agent_info:
            return None  # Agent가 없으면 None 반환

        # 3. Agent 멤버 목록 조회
        members_list = await self.repo.get_agent_members(agent_id)

        # 4. Pydantic 모델 조립 및 반환
        return AgentDetailResponse(
            agent_id=agent_info["agent_id"],
            agent_name=agent_info["agent_name"],
            status_name=agent_info["status_name"],
            dept_name=agent_info["dept_name"],
            members=members_list
        )

from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.ext.asyncio import AsyncSession
# from database import get_db_session
# from schemas import AgentDetailResponse
# from repository import AgentRepository
# from service import AgentService

router = APIRouter(prefix="/agents", tags=["Agent"])

# 의존성 주입을 위한 헬퍼 함수 (Service 생성)
def get_agent_service(session: AsyncSession = Depends(get_db_session)) -> AgentService:
    repo = AgentRepository(session)
    return AgentService(repo)

@router.get("/{agent_id}", response_model=AgentDetailResponse)
async def read_agent_detail(
    agent_id: str, 
    user_id: str,
    service: AgentService = Depends(get_agent_service)  # Service 객체 주입
):
    try:
        # 비즈니스 로직 실행은 Service에 위임
        agent_detail = await service.get_agent_detail(agent_id, user_id)
    except PermissionError as e:
        raise HTTPException(status_code=403, detail=str(e))
    
    if not agent_detail:
        raise HTTPException(status_code=404, detail="Agent를 찾을 수 없습니다.")
        
    return agent_detail
